import json # Load the prefixes, roots, and suffixes data with open('word_parts.json', 'r') as file: word_parts = json.load(file) def find_similar_meanings(word): results = { "prefixes": [], "roots": [], "suffixes": [] } for category, parts in word_parts.items(): for part, meaning in parts.items(): if part in word: results[category].append((part, meaning)) return results def main(): print("Welcome to the Simple Word Part Thesaurus!") word = input("Enter a word to search for similar meanings: ").strip().lower() results = find_similar_meanings(word) if any(results.values()): print("Found the following similar word parts:") for category, matches in results.items(): if matches: print(f"\n{category.capitalize()}: ") for match in matches: print(f"- {match[0]}: {match[1]}") else: print("No similar word parts found.") if __name__ == "__main__": main()
